# Start with the official n8n image
FROM n8nio/n8n:latest

# Switch to root to install system packages
USER root

# 1. Install Python and build dependencies (needed for Pandas/TA-Lib)
RUN apk add --update --no-cache \
    python3 \
    py3-pip \
    build-base \
    python3-dev \
    libffi-dev \
    openssl-dev \
    wget

# 2. Create a virtual environment for Python (Recommended for Alpine Linux)
ENV VIRTUAL_ENV=/opt/venv
RUN python3 -m venv $VIRTUAL_ENV
ENV PATH="$VIRTUAL_ENV/bin:$PATH"

# 3. Copy requirements and install Python libraries
COPY requirements.txt /tmp/requirements.txt
RUN pip install --upgrade pip && \
    pip install -r /tmp/requirements.txt

# 4. (Optional) Install TA-Lib (Technical Analysis Library) C-Dependencies
# Note: TA-Lib compilation on Alpine can be tricky. 
# For simplicity, we stick to the pip install above, but in prod, you compile the C-lib here.

# Switch back to the n8n user
USER node
